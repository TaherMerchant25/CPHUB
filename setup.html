Step 1: Create Database Table

Go to your Supabase SQL Editor and run this SQL:

-- Create the users table CREATE TABLE IF NOT EXISTS users ( id BIGSERIAL PRIMARY KEY, username TEXT UNIQUE NOT NULL, total INTEGER DEFAULT 0, easy INTEGER DEFAULT 0, medium INTEGER DEFAULT 0, hard INTEGER DEFAULT 0, questions TEXT[] DEFAULT '{}', created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(), updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW() ); -- Create index CREATE INDEX IF NOT EXISTS idx_users_username ON users(username); -- Create update function CREATE OR REPLACE FUNCTION update_updated_at_column() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ language 'plpgsql'; -- Create trigger DROP TRIGGER IF EXISTS update_users_updated_at ON users; CREATE TRIGGER update_users_updated_at BEFORE UPDATE ON users FOR EACH ROW EXECUTE FUNCTION update_updated_at_column(); -- Enable RLS ALTER TABLE users ENABLE ROW LEVEL SECURITY; -- Create policy DROP POLICY IF EXISTS "Enable all access for authenticated users" ON users; CREATE POLICY "Enable all access for authenticated users" ON users FOR ALL USING (true) WITH CHECK (true);
